# apt packages installed only on the laptop (gui tier).
# Firefox Developer Edition, Thunderbird Beta, WezTerm, VS Code Insiders and
# Obsidian need extra apt repos, tarballs or a direct .deb, so they are
# installed by lib/install-gui.sh instead of listed here.
# xz-utils is needed to unpack the Thunderbird Beta .tar.xz tarball.
xz-utils

# Evolution mail/calendar client, plus the Exchange Web Services connector.
# Both are plain Ubuntu universe packages, no extra repo needed.
evolution
evolution-ews

# VLC media player, also a plain Ubuntu universe package.
vlc

# Legacy FUSE2 library: AppImage-packaged apps need it to run.
libfuse2t64

# LibreOffice office suite + offline English help. The apps themselves
# (Writer/Calc/Impress/Draw/Math/Base) are hard Depends of the metapackage,
# but --no-install-recommends drops the desktop integration and the
# metric-compatible fonts, so those are named explicitly.
libreoffice
libreoffice-help-en-us
libreoffice-gnome        # native GTK file dialogs + theming
fonts-crosextra-caladea  # Cambria-metric
fonts-crosextra-carlito  # Calibri-metric
fonts-liberation         # Arial/Times/Courier-metric
